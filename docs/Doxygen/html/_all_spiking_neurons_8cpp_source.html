50void AllSpikingNeurons::advanceVertices(AllEdges &synapses, const EdgeIndexMap *edgeIndexMap)
51{
52 int maxSpikes = (int)((Simulator::getInstance().getEpochDuration()
53 * Simulator::getInstance().getMaxFiringRate()));
54
55 AllSpikingSynapses &spSynapses = dynamic_cast<AllSpikingSynapses &>(synapses);
56 // For each neuron in the network
57 for (int idx = Simulator::getInstance().getTotalVertices() - 1; idx >= 0; --idx) {
58 // advance neurons
59 advanceNeuron(idx);
60
61 // notify outgoing/incoming synapses if neuron has fired
62 if (hasFired_[idx]) {
63 LOG4CPLUS_DEBUG(vertexLogger_,
64 "Neuron: " << idx << " has fired at time: "
65 << g_simulationStep * Simulator::getInstance().getDeltaT());
66
67 assert(vertexEvents_[idx].getNumEventsInEpoch() < maxSpikes);
68
69 // notify outgoing synapses
70 BGSIZE synapseCounts;
71
72 if (edgeIndexMap != nullptr) {
73 synapseCounts = edgeIndexMap->outgoingEdgeCount_[idx];
74 if (synapseCounts != 0) {
75 int beginIndex = edgeIndexMap->outgoingEdgeBegin_[idx];
76 BGSIZE iEdg;
77 for (BGSIZE i = 0; i < synapseCounts; i++) {
78 iEdg = edgeIndexMap->outgoingEdgeIndexMap_[beginIndex + i];
79 spSynapses.preSpikeHit(iEdg);
80 }
81 }
82 }
83
84 // notify incoming synapses
85 synapseCounts = spSynapses.edgeCounts_[idx];
86 BGSIZE synapse_notified = 0;
87
88 if (spSynapses.allowBackPropagation()) {
89 for (int z = 0; synapse_notified < synapseCounts; z++) {
90 BGSIZE iEdg = Simulator::getInstance().getMaxEdgesPerVertex() * idx + z;
91 if (spSynapses.inUse_[iEdg] == true) {
92 spSynapses.postSpikeHit(iEdg);
93 synapse_notified++;
94 }
95 }
96 }
97
98 hasFired_[idx] = false;
99 }
100 }
101}

122uint64_t AllSpikingNeurons::getSpikeHistory(int index, int offIndex)
123{
124 // offIndex is a minus offset
125 // This computes the index of a spike in the past (i.e., the most recent spike,
126 // two spikes ago, etc). It starts with `spikeCountOffset_ + spikeCount_`,
127 // which I believe at the end of an epoch should be one past the end of that
128 // neuron's spikes in `spikeHistory_`. Then, the maximum number of spikes per
129 // epoch is added. Then, the `offIndex` parameter is added. The expectation
130 // is that `offIndex` will be a negative number (i.e., a spike in the past);
131 // the reason that the max spikes value is added is to prevent this from
132 // producing a negative total, so that finally taking mod max spikes will
133 // "wrap around backwards" if needed.
134
135 return vertexEvents_[index].getPastEvent(offIndex);
136}
